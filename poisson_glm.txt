

data {
  N <- dim(stars);
}

model {
  ## PRIORS
  ln_lambda_bef ~ dnorm(0,0.01);
  ln_lambda_aft ~ dnorm(0,0.01);
  lambda_bef <- exp(ln_lambda_bef);
  lambda_aft <- exp(ln_lambda_aft);
  ## LIKELIHOOD
  for(i in 1:N[1]) {
    stars[i,1] ~ dpois(lambda_aft * area);
    stars[i,2] ~ dpois(lambda_bef * area);
  }
} ## end model description
